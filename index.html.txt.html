let coins = 10;
let score = 0;
let timeLeft = 10;
let gameInterval;
let countdownInterval;
let currentGame = "";
const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
function playSound(type) {
if (!audioCtx) return;
let osc = audioCtx.createOscillator();
let gain = audioCtx.createGain();
osc.connect(gain);
gain.connect(audioCtx.destination);
if (type === 'click') {
osc.frequency.setValueAtTime(400, audioCtx.currentTime);
gain.gain.setValueAtTime(0.2, audioCtx.currentTime);
osc.start(); osc.stop(audioCtx.currentTime + 0.05);
} else if (type === 'win') {
osc.frequency.setValueAtTime(880, audioCtx.currentTime);
gain.gain.setValueAtTime(0.3, audioCtx.currentTime);
osc.start(); osc.stop(audioCtx.currentTime + 0.3);
} else if (type === 'lose') {
osc.type = 'sawtooth';
osc.frequency.setValueAtTime(120, audioCtx.currentTime);
gain.gain.setValueAtTime(0.5, audioCtx.currentTime);
osc.start(); osc.stop(audioCtx.currentTime + 0.4);
triggerShock();
} else if (type === 'tick') {
osc.frequency.setValueAtTime(900, audioCtx.currentTime);
gain.gain.setValueAtTime(0.05, audioCtx.currentTime);
osc.start(); osc.stop(audioCtx.currentTime + 0.03);
}
}
function triggerShock() {
document.getElementById('main-body').classList.add('shock-effect');
setTimeout(() => { document.getElementById('main-body').classList.remove('shock-effect'); }, 200);
if (navigator.vibrate) navigator.vibrate(300);
}

document.getElementById('user-coins').innerText = coins;

}

updateCoinDisplay();
toggleShop(false);
}

/* RASIO AKURASI TINGGI SETINGAN BANDAR: 75% LAHIRKAN KEKALAHAN, 25% KEMENANGAN */
function getSettinganResult() {
let chance = Math.random() * 100;
return chance < 25;
}
function startTimer(seconds) {
timeLeft = seconds;
document.getElementById('timer-display').innerText = "WAKTU: " + timeLeft;
countdownInterval = setInterval(() => {
timeLeft--;
document.getElementById('timer-display').innerText = "WAKTU: " + timeLeft;
if (timeLeft <= 3 && timeLeft > 0) playSound('tick');
if (timeLeft <= 0) {
clearInterval(countdownInterval);
gameOver("WAKTU HABIS!");
}
}, 1000);
}
function gameOver(reason) {
clearInterval(gameInterval);
clearInterval(countdownInterval);
playSound('lose');
document.getElementById('timer-display').innerText = "GAME OVER";
document.getElementById('game-content').innerHTML = `
<div style="color:#ff0055; font-size:22px; font-family:'Orbitron'; font-weight:bold; margin-bottom:10px;">💥 ${reason} 💥</div>
<div class="score-board">Skor Akhir: ${score}</div>

}

function moveWildWestTarget() {
const tar = document.getElementById('target');
if(!tar) return;
tar.style.left = Math.floor(Math.random() * 80) + '%';
tar.style.top = Math.floor(Math.random() * 70) + 'px';
let speed = Math.max(300, 700 - (score * 80));
gameInterval = setTimeout(moveWildWestTarget, speed);
}
function hitWildWestTarget() {

playSound('click');

clearInterval(countdownInterval);
clearInterval(gameInterval);

document.getElementById('game-content').innerHTML = `<h2 style='color:#4caf50;'>🤠 BURONAN TERTANGKAP! Menang!</h2>`;
}
}
